% A
\begin{align*}
	&\sum{\vec{F}_x} := 0 = A_x + N_2 + {N_1}_x\\
	&\sum{\vec{F}_y} := 0 = A_y - {N_1}_y
\end{align*}

\begin{center}
	$\tan \alpha = \frac{c}{a} \Rightarrow \alpha = \pnum{\alfa}$\textdegree
\end{center}
\begin{align*}
	&{N_1}_y = A_y = \pnum{\Nay} \\
	&\mathbf{N_1} = \frac{{N_1}_y}{\sin \alpha} = \mathbf{\pnum{\Na}} \\
	&{N_1}_x = N_1 \times \cos \alpha = \pnum{\Nax}
\end{align*}
